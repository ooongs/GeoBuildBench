# 회전을 이용한 정사각형 예제
# 한 점을 90도씩 회전시켜 정사각형을 만듭니다

# 중심점과 시작점 정의
point : 150 150 -> Center
point : 200 150 -> A

# 90도(π/2) 회전으로 나머지 점들 생성
rotate : A 90° Center -> B
rotate : B 90° Center -> C
rotate : C 90° Center -> D

# 선분으로 연결 (polygon 명령어 사용 금지!)
segment : A B -> AB
segment : B C -> BC
segment : C D -> CD
segment : D A -> DA
